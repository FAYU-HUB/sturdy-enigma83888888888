Pilih foto, lalu klik proses untuk menempelkan angka 70 yang transparan.

const preview = document.getElementById('preview'); const downloadBtn = document.getElementById('downloadBtn'); processBtn.addEventListener('click', () => { if (!upload.files[0]) return alert('Pilih foto dulu!'); const reader = new FileReader(); reader.onload = (event) => { const img = new Image(); img.onload = () => { canvas.width = img.width; canvas.height = img.height; ctx.drawImage(img, 0, 0); const imageData = ctx.getImageData(0, 0, canvas.width, canvas.height); const data = imageData.data; const secretMessage = "HIDDEN-70"; for (let i = 0; i < secretMessage.length; i++) { const charCode = secretMessage.charCodeAt(i); data[i * 4] = (data[i * 4] & 0xFE) | (charCode % 2); } ctx.putImageData(imageData, 0, 0); const dataUrl = canvas.toDataURL('image/png'); preview.src = dataUrl; preview.style.display = 'block'; downloadBtn.href = dataUrl; downloadBtn.download = 'hidden_70_image.png'; downloadBtn.style.display = 'block'; }; img.src = event.target.result; }; reader.readAsDataURL(upload.files[0]); });
